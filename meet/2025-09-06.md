## pygame

```python
import sys

import pygame

pygame.init()

clock = pygame.time.Clock()

screen_w, screen_h = 640, 480

screen = pygame.display.set_mode((screen_w, screen_h))

ball_a = pygame.Rect(0, 0, 30, 30)

ball_b = pygame.Rect(0, screen_h - 30, 30, 30)

while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            sys.exit()

    if ball_a.colliderect(ball_b):
        screen.fill('Yellow')
    else:
        screen.fill('Black')

    pygame.draw.rect(screen, 'Blue', ball_a)

    pygame.draw.rect(screen, 'Red', ball_b)

    ball_a.left += 4

    if ball_a.right >= screen_w:
        ball_a.left = 0

    ball_b.top += 7

    if ball_b.top >= screen_h:
        ball_b.top = 0

    pygame.display.update()

    clock.tick(60)
```
